[docs]    def update_incumbent(
        self,
        objective_value: float,
        best_bound: float,
        solution: List[Tuple[Var, float]],
    ) -> List[Tuple[Var, float]]:
        dict_solution = {
            "obj": objective_value,
            "best_bound": best_bound,
            "solution": {var[0].name: var[1] for var in solution},
        }
        self._solution_store += [dict_solution]
        return solution
